33 QByteArray translate(
34  const QByteArray &array,
35  const Info &fromInfo,
36  const Info &toInfo
37  ) noexcept(false)
38 {
39  // バイト配列の長さ
40  auto maxSize = array.length();
41 
42  // バイト配列の長さによって変換する単位バイトを決定
43  // 16KB以上であれば8KB単位、より小さければ1KB単位
44  auto CHUNK_SIZE = (maxSize >= KBtoB<int>(16)) ? KBtoB<int>(8) : KBtoB<int>(1);
45 
46  // オフセット位置
47  int offset = 0;
48 
49  // 変換結果の格納先
50  QByteArray result;
51 
52  // オフセット位置が長さを超えるまでループ
53  while (offset < maxSize) {
54 
55  // 変換対象が単位バイトを超えた場合は単位バイト、
56  // 超えていなければ変換対象全体の長さをチャンクサイズにする。
57  auto chunkSize = std::min<int>(maxSize - offset, CHUNK_SIZE);
58 
59  // チャンクサイズが文字の途中にならないように調整
60  chunkSize = fromInfo.adjustedByteSize(
61  array.constData() + offset,
62  chunkSize
63  );
64 
65  // チャンクサイズが0になったら終了
66  if (chunkSize <= 0) break;
67 
68  // バイト配列からチャンクサイズ分だけ取得
69  auto chunk = array.mid(offset, chunkSize);
70 
71  // 変換したバイト配列を格納するバッファを準備
72  auto bufSize = chunkSize * ratio;
73  QByteArray buffer(bufSize, '\0');
74 
75  // 変換する
76  auto len = static_cast<WORD>(bufSize);
77  Status status = NLS_translate(
78  reinterpret_cast<BYTE*>(chunk.data()),
79  static_cast<WORD>(chunkSize),
80  reinterpret_cast<BYTE*>(buffer.data()),
81  &len,
82  mode,
83  toInfo.get()
84  );
85 
86  // エラーなら例外をスロー
87  if (status.hasError()) throw status;
88 
89  // 変換したバイト配列を格納先に追加
90  result.append(buffer.left(static_cast<int>(len)));
91 
92  // オフセット位置をチャンクサイズ分移動
93  offset += chunkSize;
94  }
